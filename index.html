Carregar as tarefas assim que a página abrir window.onload = function() { const tarefasSalvas = JSON.parse(localStorage.getItem("minhasTarefas")) || []; tarefasSalvas.forEach(t => criarElementoTarefa(t.texto, t.concluida)); }; function adicionar() { let input = document.getElementById("tarefa"); let texto = input.value; if (texto.trim() === "") return; criarElementoTarefa(texto, false); salvarNoStorage(); // Salva após adicionar input.value = ""; } // Função auxiliar para criar o HTML da tarefa function criarElementoTarefa(texto, concluida) { let li = document.createElement("li"); li.className = "list-group-item d-flex justify-content-between align-items-center border-0 mb-2 bg-light rounded-3 item-tarefa"; let classeConcluida = concluida ? "concluida" : ""; li.innerHTML = ` ${texto} `; document.getElementById("lista").appendChild(li); } function remover(botao) { botao.parentElement.remove(); salvarNoStorage(); // Salva após remover } function concluir(texto) { texto.classList.toggle("concluida"); salvarNoStorage(); // Salva após marcar como feita } // 2. Função que varre a lista e salva no LocalStorage function salvarNoStorage() { const listaLi = document.querySelectorAll("#lista li"); const tarefas = []; listaLi.forEach(li => { tarefas.push({ texto: li.querySelector("span").innerText, concluida: li.querySelector("span").classList.contains("concluida") }); }); localStorage.setItem("minhasTarefas", JSON.stringify(tarefas)); }
